Program -> Statements
Statements -> Statement Statements | Îµ
Statement -> VarDeclaration
           | Assignment
           | PrintStatement
           | CompoundStatement
           | IfStatement
           | WhileStatement
           | ForStatement
VarDeclaration -> 'var' IDENTIFIER '=' Expression ';'
Assignment -> IDENTIFIER '=' Expression ';'
PrintStatement -> 'print' Expression ';'
CompoundStatement -> '{' Statements '}'
IfStatement -> 'if' '(' Expression ')' CompoundStatement
             | 'if' '(' Expression ')' CompoundStatement 'else' CompoundStatement
WhileStatement -> 'while' '(' Expression ')' CompoundStatement
ForStatement -> 'for' '(' Assignment ';' Expression ';' Assignment ')' CompoundStatement
Expression -> SimpleExpression (('==' | '!=' | '<' | '<=' | '>' | '>=') SimpleExpression)*
SimpleExpression -> Term (('+' | '-' | 'add' | 'sub') Term)*
Term -> Factor (('*' | '/' | 'mult' | 'div') Factor)*
Factor -> INTEGER
        | FLOAT
        | STRING
        | BOOLEAN
        | IDENTIFIER
        | '(' Expression ')'
